WorkerQueue

Skip to end of metadata
  • Created by Mike Austin, last modified on Sep 28, 2016
Go to start of metadata

Summary

WorkerQueue is a CPU and memory friendly alternative to starting lots of short-lived coroutines that work on data.  It utilizes object-pooling to achieve zero allocation operation, and supports nested blocking workers similar to nested coroutines.

So instead of starting many coroutines, you start one and add data to the queue for it to process.  Potential candidates are PrefabLoader, MaterialLoader, etc.

Example Worker

?
// An item struct used to pass data to the workers
public struct PrefabLoaderItem {
 
    public GameObject Parent { get; private set; }
    public string PrefabName { get; private set; }
 
    public PrefabLoaderItem(GameObject parent, string prefabName) {
        this.Parent     = parent;
        this.PrefabName = prefabName;
    }
 
}
 
// A worker class that takes PrefabLoaderItem as an argument to Configure()
public class PrefabLoaderWorker : ConfigurableWorker<PrefabLoaderItem> {
 
    private long _count;
    private Object _request;
    private PrefabLoaderItem _item;
 
    // Called once to pass worker item to worker and to do any other initialization.
    public override void Configure(PrefabLoaderItem item) {
        this._count   = 0;
        this._request = AssetManager.Instance.LoadAsset(item.PrefabName + ".prefab", downloadIfNecessary: true);
        this._item    = item;
    }
 
    // Called every frame while true is returned. Called by WorkerQueue.ProcessQueue().
    public override bool MoveNext() {
        if (this._count-- > 0) {
            return true;
        }
 
        return false;
    }
 
    // Called once when Run() returns false. The worker is returned to the worker pool.
    public override void Reset() {
        this._request = null;
    }
 
}

Example Usage

?
public class PrefabLoader {
 
    private WorkerQueue<PrefabLoaderWorker, PrefabLoaderItem> _workerQueue;
 
    void PrefabLoader() {
        // Create a new WorkerQueue that creates (and pools) PrefabLoaderWorkers, with an initial pool capacity of 100 items.
        this._workerQueue = new WorkerQueue<PrefabLoaderWorker, PrefabLoaderItem>(initialPoolCapacity: 100);
        CoroutineHelper.StartCoroutine(ProcessWorkerQueue());
 
    }
 
    void InstantiateObject(GameObject parentObject, string prefabName) {
        // Create a new item and add it to the worker queue. The item will be processed on the next frame.
        var item = new PrefabLoaderItem(parent: someObject, prefabName: "cityHallDialog");
        workerQueue.StartWorker(item);
    }
 
    IEnumerator ProcessWorkerQueue() {
        while (true) {
            _workerQueue.ProcessQueue();
 
            yield return null;
        }
    }
}
